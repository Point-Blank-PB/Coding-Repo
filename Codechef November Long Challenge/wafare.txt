#include<stdio.h>
int main()
{
int n,t;
scanf("%d",&t);
while(t!=0)
{
scanf("%d",&n);
int pk[n][10];
for(int i=0;i<n;i++)
{
for(int j=0;j<10;j++)
{scanf("%d",&pk[i][j]);}
}
//compare....
for(int i=0;i<n-1;i++)
{
for(int j=0;j<10;j++)
{if(pk[i][j]==1 && pk[i+1][j]==1)
{pk[i+1][j]=0;}
else if(pk[i][j]==0 && pk[i+1][j]==0)
{pk[i+1][j]=0;}
else if(pk[i][j]==1 && pk[i+1][j]==0)
{pk[i+1][j]=1;}
else if(pk[i][j]==0 && pk[i+1][j]==1)
{pk[i+1][j]=1;}
}
}
int c=0;
for(int j=0;j<10;j++)
{if(pk[n-1][j]==1)
{c++;}
}
printf("%d",c);}
return 0;
}
